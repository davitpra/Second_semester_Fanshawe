<html>
<head>
    <title>Bouncing Ball Demo</title>

    <style type="text/css">

        #ball{
            position: absolute;
            height: 32px;
            width: 32px;
            top:0;
            left:0;
        }

        #ball_container{
            position: absolute;
            top: 10px;
            left: 10px;
            width: 800px;
            height: 600px;
            background-color: darkgrey;
        }
    </style>

</head>
<body onload="main();">

<script type="text/JavaScript">
    function main(){

        function Ball(position){
            this.image = document.getElementById("ball");

            this.position = {
                x : position.x,
                y : position.y
            };

            this.direction = {
                x: 1,
                y: 1
            };
        }

        var ball = new Ball({x:0,y:0});


        function moveBall(){
            ball.position.x += ball.direction.x;
            ball.position.y += ball.direction.y;
        }

        function update(){
            //reposition ball
            ball.image.style.top = ball.position.y + "px";
            ball.image.style.left = ball.position.x + "px";

            //are we at the top?
            if(ball.position.y === 0){
                ball.direction.y = 1;
            }

            //are we at the bottom?
            if(ball.position.y === 575){
                ball.direction.y = -1;
            }

            //are we at the left?
            if(ball.position.x === 0){
                ball.direction.x = 1;
            }

            //are we at the right edge?
            if(ball.position.x === 775){
                ball.direction.x = -1;
            }

            moveBall();

            setTimeout(update, 16);
        }

        update();
    }
</script>

<div id="ball_container">
    <img id="ball" src="images/basketball.png" alt="basketball" />
</div>

</body>
