<html>
<head>
    <title>Multiple Bouncing Ball Demo</title>

    <style type="text/css">

        .ball{
            position: absolute;
            height: 32px;
            width: 32px;
            top:0;
            left:0;
        }

        #ball_container{
            position: absolute;
            top: 10px;
            left: 10px;
            width: 800px;
            height: 600px;
            background-color: darkgrey;
        }
    </style>

</head>

<body onload="main();">

<script type="text/JavaScript">
    function Ball(position){
        //document.getElementById("ball");

        //this creates the actual image in javascript
        this.image = document.createElement("img");
        this.image.src = "images/basketball.png";
        this.image.classList.add("ball");

        //add the image to the ball container
        document.getElementById("ball_container").appendChild(this.image);

        this.position = {
            x : position.x,
            y : position.y
        };

        this.direction = {
            x: Math.floor(Math.random() * 2) % 2 === 0 ? 1 : -1,
            y: Math.floor(Math.random() * 2) % 2 === 0 ? 1 : -1
        };
    }

    Ball.prototype.move = function(){
        this.position.x += this.direction.x;
        this.position.y += this.direction.y;
    }

    function main(){
        let numberOfBalls = 5;
        let balls = [];
        for(let i=0; i<numberOfBalls; ++i){
            balls.push(new Ball({x: (i * 32) + 100,y:100}));
        }

        function update(){
            //reposition ball
            for(let i=0; i<balls.length; ++i){
                balls[i].image.style.top = balls[i].position.y + "px";
                balls[i].image.style.left = balls[i].position.x + "px";

                //are we at the top?
                if(balls[i].position.y === 0){
                    balls[i].direction.y = 1;
                }

                //are we at the bottom?
                if(balls[i].position.y === 575){
                    balls[i].direction.y = -1;
                }

                //are we at the left?
                if(balls[i].position.x === 0){
                    balls[i].direction.x = 1;
                }

                //are we at the right edge?
                if(balls[i].position.x === 775){
                    balls[i].direction.x = -1;
                }

                balls[i].move();
            }

            setTimeout(update, 16);
        }

        update();
    }
</script>

<div id="ball_container">
</div>

</body>
